SELECT venue_address, event_id, author_name, paper_name, reviewer_name, chair_name FROM Venue, Venue_info, Author, Author_info, Paper, Paper_info, Reviewer, Reviewer_info, Chair WHERE Venue.venue_id = Venue_info.venue_id AND Venue_info.venue_id = Author.venue_id AND Author.author_id = Author_info.author_id AND Author_info.author_id = Paper.author_id AND Paper.paper_id = Paper_info.paper_id AND Paper.paper_id = Reviewer.paper_id AND Reviewer.reviewer_id = Reviewer_info.reviewer_id AND Reviewer_info.reviewer_id = Chair.reviewer_id;
