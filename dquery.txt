SELECT event_name, venue_address, author_name, paper_name FROM Event, Venue, Author, Paper WHERE Venue.event_id = Event.event_id AND Venue.venue_id = Author.venue_id AND Author.author_id = Paper.author_id;
SELECT venue_address, event_id, author_name, paper_name, reviewer_name, chair_name FROM Venue, Author, Paper, Reviewer, Chair WHERE Author.venue_id = Venue.venue_id AND Paper.author_id = Author.author_id AND Reviewer.paper_id = Paper.paper_id AND Chair.reviewer_id = Reviewer.reviewer_id;
SELECT event_date, venue_address, author_company, editor_id, reviewer_address, responsibility FROM Event, Venue, Author, Paper, Reviewer, Chair WHERE Event.event_id = Venue.event_id AND Venue.venue_id = Author.venue_id AND Author.author_id = Paper.author_id AND Reviewer.paper_id = Paper.paper_id AND Reviewer.reviewer_id = Chair.reviewer_id;
